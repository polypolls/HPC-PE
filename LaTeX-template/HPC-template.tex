std::vector<float> MLPMath::applyReLU(std::vector<float>& input) {

    std::vector<float> result(input.size());

    for (std::size_t i = 0; i < input.size(); i++) {

        if (input[i] >= 0) {  
            result[i] = input[i];
        }
        else {
            result[i] = 0.0f;
        }
    }

    return result;
}
\end{lstlisting}
